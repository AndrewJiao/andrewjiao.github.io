-- 把db1库和t1表的查询字段id，和name权限给用户
grant select(id,name) on db1.tb1 to 'abc'@'%';

###外键约束(外键特点4个)

  • 特点
    1. 引用主键的值
    2. 允许重复，也允许null值
    3. 自动会创建索引
    4. *外键会降低表的访问效率,在大型互联网应用中,数据库表一般都不添加外键约束,会用程序来保证数据的正确性
  • 建表时添加外键
    1. 创建学生表；（两个）分别是班级和学生（id，name）
    2. 修改学生表，添加班级id外键引用班级的主键
  • 修改表示添加外键（格式同上）
-- 创建学生表
drop table if exists xuesheng;
create table xuesheng(
id int primary key auto_increment,
name varchar(20)
)engine=innodb charset=utf8
-- 创建班级表

create table banji(
id int,
name varchar(20)
)engine=innodb charset=utf8;
-- 修改学生表,添加banji_id外键,引用班级的主键

alter table banji modify id int auto_increment;
alter table xuesheng add foreign key(banji_id) references banji(id);

事物

  • 事务由多个数据操作(增删改查)组合

  • 事务是数据操作的最小单元

  • 事务中所有操作全部成功,事务整体成功

  • 当事务中一项操作失败,事务整体失败

    • 例如,游戏中 A 账户向 B 账户转账

      1. A 更新,减掉金额

      2. B 更新,加金额失败(断电),转账事务要整体失败,回退到以前的状态

    • 事务特性:ACID

      • A - 原子性 Atomic
      • C - 一致性 Consistency
        1. 转账前 a+b = 100
        2. 转帐后 a+b = 100
      • I - 隔离性 Isolation
        1. 一个事物进行中时
        2. 另一事物不能操作数据
      • D - 持久性 Durancy
        1. 提交事务之后，
        2. 数据持久生效

事物隔离

  1. 事务完全隔离,一个事务完成,另一个事务才能执行,效率低
  2. 事务不隔离,会有数据访问冲突,数据不安全
  3. 数据库当中,可以设置事务的隔离性,来兼顾安全和性能
  • 脏读(一般不允许发生)：一个会话未提交的数据,被另一个会话读取
  • 不可重复读：同一行数据,再次执行相同查询,与第一次查询结果不一致
  • 幻读
    1. 其它会话新插入并提交的数据,查询不到
    2. 其它会话删除并提交的数据,仍然能查询到

事物隔离级别

  1. set tx_isolation=’READ-UNCOMMITTED’;脏读,不可重复读,幻读
  2. set tx_isolation=’read-committed’;不可重复读,幻读
  3. set tx_isolation=’repeatable-read’;幻读
  4. set tx_isolation=’serializable’;

create procedure p4(in n int)
begin
    declare i int default 0;
    while i<n do
        insert into t values(i);
        set i=i+1;
    end while;
    set i=0;
    lp:loop
        insert into t values(i);
        set i=i+1;
        if i>=n then
            leave lp;
        end if;
    end loop;
    set i=0;
    repeat
        insert into t values(i);
        set i=i+1;
    until i>=n end repeat;
end;

-- 磁盘表（字段为id的主键和c）
create table tb(
id bigint primary key auto_increment,
d varchar(10)
)engine=innodb charset=utf8;
-- 批量产生数据的存储过程
drop procedure if exists gen_data;
create procedure gen_data(in n int)
begin
    -- 产生字符串并向表内插入数据
    declare i int default 0;
    repeat
        set i=i+1;
        insert into tm(c) values(fn2(3,6));
    until i>=n end repeat;
    insert into tb(d) select c from tm;
    delete from tm;
end;
//
call gen_data();

触发器

  • 插入，修改，删除时，可以自动触发的一段代码

  • 可以在操作之前或者之后触发

  • 一共有六种触发器

    1. before insert
    2. before update
    3. before delete
    4. after insert
    5. after update
    6. after delete
    • 两个隐含对象
      • new - 新数据
      • oud - 旧数据
      • insert
        • new 插入新数据
        • old 没有
      • update
        • new 修改之后的新数据
        • old 修改之前的旧数据
      • delete
        • new 没有
        • old 被删除的旧数据
      • 访问新的数据行或旧的数据行
        • new.employee_id;
        • old.first_name;

-- 新建tb1表(主键id，字段c，更新时间，创建时间)
drop table if exists tb1;//
create table tb1(
id int primary key auto_increment,
c varchar(255),
update_time datetime,
create_time datetime
)engine=innodb charset=utf8;//
-- 新建触发器,插入数据时
-- 自动为created和updated填入当前时间
drop trigger if exists tb1_before_insert//
create trigger tb1_before_insert before insert on tb1 for each row
begin
    set new.update_time=now();
    set new.create_time=now();
end//
insert into tb1(c) values('a');

select * from tb1;//
